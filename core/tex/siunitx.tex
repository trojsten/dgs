\sisetup{
    detect-all              = true,                         % keep formatting around
    separate-uncertainty    = true,                         % 7.2 ± 0.5
    multi-part-units        = single,                       % 12.3 ± 0.2 kg, do not repeat "kg"
    per-mode                = reciprocal,                   % `symbol` for "m/s", `reciprocal` for "ms^{-1}"
    group-separator         = {\,},
    group-minimum-digits    = 5,
    inter-unit-product      = {\,},                         % add a very thin space between units
    input-decimal-markers   = {.},
    exponent-product        = \cdot,                        % \times for 5×10^7, \cdot for 5.1O^7
    number-unit-product     = {\ },                         % a non-breakable space between numbers and units
    output-decimal-marker   = {\text{.}},                   % decimal dot
    range-units             = single,
    range-phrase            = {\text{ -- }},                % 5 -- 10 m/s
    list-units              = single,                       %
    list-pair-separator     = {\text{\ and\ }},             % 5 a 10 m/s
    list-final-separator    = {\text{\ and\ }},             % 5, 7 a 10 m/s
    retain-explicit-plus    = true,
    math-ohm                = \Omega,
    text-ohm                = \ensuremath{\Omega},
    math-micro              = \text{μ},
    text-micro              = μ,
}
