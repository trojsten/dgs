\sisetup{
    detect-all                  = true,                         % keep formatting around
    separate-uncertainty        = true,                         % 7.2 ± 0.5
    multi-part-units            = brackets,                     % 12.3 ± 0.2 kg, do not repeat "kg"
    per-mode                    = reciprocal,                   % `symbol` for "m/s", `reciprocal` for "ms^{-1}"
    group-separator             = {\,},
    group-minimum-digits        = 5,
    inter-unit-product          = {},                           % nothing between units
    input-decimal-markers       = {.},
    exponent-product            = \cdot,                        % \times for 5×10^7, \cdot for 5.1O^7
    number-unit-product         = {\ },                         % a non-breakable space between numbers and units
    range-units                 = single,
    range-phrase                = {\text{ -- }},                % 5 -- 10 m/s
    list-units                  = single,                       %
    retain-explicit-plus        = true,
}
