\NewDocumentCommand{\Derivative@}{d<> o D<>{\diff} m m}{%
    \SetDerivativeFrac@{#1}
    \IfNoValueTF{#2}{%
        \frac@{#3#4}{\cdiff{#5}{#3}}%
    }{%
        \frac@{#3^{#2}#4}{\cdiff{#5}{#3}}%
    }%
}

\NewDocumentCommand{\DerivativeParentheses@}{d<> o D<>{\diff} m m}{%
    \SetDerivativeFrac@{#1}
    \IfNoValueTF{#2}{%
        \frac@{#3}{#3#5}\left(#4\right)%
    }{%
        \frac@{#3^{#2}}{#3{#5}^{#2}}\left(#4\right)%
    }%
}

\ExplSyntaxOn

    }{\frac}}%

        \frac@{#3}{#3#4}%
    }{%

    }%
}
